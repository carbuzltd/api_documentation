accounts

Methods

(static) accounts_load(opts)

Get all the accounts containing a selected string specified in the search. If search is empty, it returns all the accounts present in the database.

Example
const axios = require('axios');
axios({
  method: 'post',
  url: url + 'accounts_load',
  responseType: 'json',
  data: {
    sid: '[api_key]',   //The string your admin gave to you
    search: '12',       //Search for account containing this string
  }
})
.then(function (response) {

  console.log(response.data);  
  //Result:
  //[
  //   {
  //     mng_advisor_fee: 0,
  //     perf_advisor_fee: 0,
  //     key: '123456789',
  //     name: 'James Bond',
  //     key_advisor: null,
  //     subaccounts: '',
  //     mng_fee: 100,
  //     perf_fee: 10,
  //     limits: { class: 'UCIS+FundsLim' },
  //     ccy: 'GBP',
  //     key_clearer: 'VIRTUALE',
  //     key_client: null,
  //     last_account_value: 1324439.83,
  //     pnl_qtd: -20659.53,
  //     avg_amount_qtd: 1345099.36,
  //     last_date: '2021-02-02',
  //     limits_class: 'UCIS+FundsLim',
  //     estimate_mng_fee_qtd: 13450.993600000002,
  //     estimate_perf_fee_qtd: -20.65953
  //   }
  // ]
})
.catch(function (error) {
  // handle error
  console.log(error);
})
