reports

String helper methods

Methods

(static) reports_portfolioPdf(opts)

It generates a pdf report with the information specified for a portfolio at a given date.

Example
const axios = require('axios');
axios({
  method: 'post',
  url: url + 'reports_portfolioPdf',
  responseType: 'json',
  data: {
    sid: '[your_api_key]',      //The string your admin gave to you
    key_portfolio: '123456789', //Portfolio considered in the analysis
    toCcy: 'EUR',               //Result will be in this currency
    endDate: '2020-01-03'       //Result considering trades till this date
  }
})
.then(function (response) {
  
  console.log(response.data) 
  //Result:
  //{
  //  content: '...', //Pdf report encoded in base64
  //  fileName: '20200103__123456789__James_Bond.pdf',
  //  type: 'application/pdf'
  //}
})
.catch(function (error) {
  // handle error
  console.log(error);
})
